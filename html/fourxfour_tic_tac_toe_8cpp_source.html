125bool FourxFour_Board::checking_win( char player) {
126 for(int i = 0; i < 4; i++) {
127 for(int j = 0; j < 2; j++) {
128 if(board[i][j] == player && board[i][j+1] == player && board[i][j+2] == player)
129 return true;
130 if(board[j][i] == player && board[j+1][i] == player && board[j+2][i] == player)
131 return true;
132 }
133 }
134 int diagonals[4][3][2] = {
135 {{0,0},{1,1},{2,2}}, {{0,1},{1,2},{2,3}},
136 {{1,0},{2,1},{3,2}}, {{1,1},{2,2},{3,3}}
137 };
138 for(auto& d : diagonals) {
139 if(board[d[0][0]][d[0][1]] == player &&
140 board[d[1][0]][d[1][1]] == player &&
141 board[d[2][0]][d[2][1]] == player)
142 return true;
143 }
144 return false;
145}

162int FourxFour_Board::minimax(char ai, char human, bool isMaximizingPlayer, int alpha, int beta, int depth) {
163 int score = evaluate1(ai, human);
164 if (score != 0) return score;
165 if (depth == MAX_DEPTH) {
166 return evaluation2(ai, human);
167 }
168 char current_sym = isMaximizingPlayer ? ai : human;
169 vector<FourxFour_Move*> possible_moves = get_all_valid_moves(current_sym);
170 if (possible_moves.empty()) {
171 return 0;
172 }
173 int best_score = isMaximizingPlayer ? -100000 : 100000;
174 for (size_t i = 0; i < possible_moves.size(); ++i) {
175 FourxFour_Move* move = possible_moves[i];
176 update_board(move);
177 int current_score = minimax(ai, human, !isMaximizingPlayer, alpha, beta, depth + 1);
178 board[move->getNewX()][move->getNewY()] = blank_symbol;
179 board[move->get_x()][move->get_y()] = current_sym;
180 if (isMaximizingPlayer) {
181 best_score = max(best_score, current_score);
182 alpha = max(alpha, best_score);
183 } else {
184 best_score = min(best_score, current_score);
185 beta = min(beta, best_score);
186 }
187 delete move;
188 if (beta <= alpha) {
189 for (size_t j = i + 1; j < possible_moves.size(); ++j) {
190 delete possible_moves[j];
191 }
192 return best_score;
193 }
194 }
195 return best_score;
196}

237FourxFour_Move* FourxFour_Board::best_move(char ai, char human) {
238 int bestVal = -100000;
239 FourxFour_Move* final_best_move = nullptr;
240 int alpha = -100000;
241 int beta = 100000;
242 const int START_DEPTH = 0;
243 vector<FourxFour_Move*> possible_moves = get_all_valid_moves(ai);
244 if (possible_moves.empty()) return nullptr;
245 vector<FourxFour_Move*> moves_to_delete;
246 for (FourxFour_Move* move : possible_moves) {
247 update_board(move);
248 int moveVal = minimax(ai, human, false, alpha, beta, START_DEPTH + 1);
249 board[move->getNewX()][move->getNewY()] = blank_symbol;
250 board[move->get_x()][move->get_y()] = ai;
251 if (moveVal > bestVal) {
252 bestVal = moveVal;
253 if (final_best_move) {
254 moves_to_delete.push_back(final_best_move);
255 }
256 final_best_move = move;
257 } else {
258 moves_to_delete.push_back(move);
259 }
260 alpha = max(alpha, bestVal);
261 }
262 for (auto move : moves_to_delete) {
263 delete move;
264 }
265 return final_best_move;
266}

272vector<FourxFour_Move*> FourxFour_Board::get_all_valid_moves(char player_symbol) {
273 vector<FourxFour_Move*> moves;

275 for (int c = 0; c < columns; c++) {
276 if (board[r][c] == player_symbol) {
277 vector<pair<int, int>> adj_empty_cells = get_adjacent(r, c);
278 for (auto& dest : adj_empty_cells) {
279 moves.push_back(new FourxFour_Move(r, c, dest.first, dest.second, player_symbol));
280 }
281 }
282 }
283 }
284 return moves;
