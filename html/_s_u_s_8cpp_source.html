82int SUS::count_SUS() {
83 int count = 0;
84
85 // 1. Check ALL horizontal S-U-S patterns
86 for (int row = 0; row < 3; row++) {
87 if (board[row][0] == 'S' &&
88 board[row][1] == 'U' &&
89 board[row][2] == 'S') {
90 count++;
91 }
92 }
93
94 // 2. Check ALL vertical S-U-S patterns
95 for (int col = 0; col < 3; col++) {
96 if (board[0][col] == 'S' &&
97 board[1][col] == 'U' &&
98 board[2][col] == 'S') {
99 count++;
100 }
101 }
102
103 // 3. Check main diagonal (top-left to bottom-right)
104 if (board[0][0] == 'S' &&
105 board[1][1] == 'U' &&
106 board[2][2] == 'S') {
107 count++;
108 }
109
110 // 4. Check anti-diagonal (top-right to bottom-left)
111 if (board[0][2] == 'S' &&
112 board[1][1] == 'U' &&
113 board[2][0] == 'S') {
114 count++;
115 }
116
117 return count;
