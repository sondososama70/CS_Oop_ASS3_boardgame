202bool UltimateBoard::check_win_condition(const vector<vector<char>>& board, char symbol) const {
203 for (int i = 0; i < 3; i++) {
204 if (board[i][0] == symbol && board[i][1] == symbol && board[i][2] == symbol)
205 return true;
206 }
207
208 for (int j = 0; j < 3; j++) {
209 if (board[0][j] == symbol && board[1][j] == symbol && board[2][j] == symbol)
210 return true;
211 }
212 if (board[0][0] == symbol && board[1][1] == symbol && board[2][2] == symbol)
213 return true;
214 if (board[0][2] == symbol && board[1][1] == symbol && board[2][0] == symbol)
215 return true;
