145int pyramidTicTacToe_board::minimax(pyramidTicTacToe_board* board, int nOfMoves, bool isMax, char ai, char human) {
146
147 int score = board->evaluate(ai, human);
148 if (score == 10) return score - nOfMoves;
149 if (score == -10) return score + nOfMoves;
150 if (board->is_full()) return 0;
151 if (isMax) {
152 int best = -1000;
153 for (int r = 0; r < board->get_rows(); r++) {
154 for (int c = 0; c < board->get_columns(); c++) {
155 if (board->is_valid_move(r, c)) {
156 board->set_cell(r, c, ai);
157 int score = minimax(board, nOfMoves + 1, false, ai, human);
158 best = max(best, score);
159 board->set_cell(r, c, '.');
160 }
161 }
162 }
163 return best;
164 }
165
166 else {
167 int best = 1000;
168 for (int r = 0; r < board->get_rows(); r++) {
169 for (int c = 0; c < board->get_columns(); c++) {
170 if (board->is_valid_move(r, c)) {
171 board->set_cell(r, c, human);
172 int score = minimax(board, nOfMoves + 1, true, ai, human);
173 best = min(best, score);
174 board->set_cell(r, c, '.');
175 }
176 }
177 }
178 return best;
179 }

185pair<int,int>pyramidTicTacToe_board:: best_move(pyramidTicTacToe_board* board, char ai, char human) {
186 int bestVal = -1000;
187 pair<int,int> best = {-1, -1};
188 for (int r = 0; r < board->get_rows(); r++) {
189 for (int c = 0; c < board->get_columns(); c++) {
190 if (board->is_valid_move(r, c)) {
191 board->set_cell(r, c, ai);
192 int moveVal = minimax(board, 1, false, ai, human);
193 board->set_cell(r, c, '.');
194 if (moveVal > bestVal) {
195 bestVal = moveVal;
196 best = {r, c};
197 }
198 }
199 }
200 }
201 return best;
202}
